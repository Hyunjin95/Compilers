170  //
171  // module ::= "module" ident ";" varDeclaration { subroutineDecl }
172  // "begin" statSequence "end" ident ".".
173  //
174

176  CToken module_name_check;

178 
179  Consume(tModule);
180  Consume(tIdent, &module_name);
181  Consume(tSemicolon);
182

184 
185  InitSymbolTable(m->GetSymbolTable());
186 
187  // varDeclaration
188  if(_scanner->Peek().GetType() == tVar) {
189  Consume(tVar);
190  varDecl(m);
191  Consume(tSemicolon);
192 
193  while(1) {
194  if(_scanner->Peek().GetType() == tIdent) {
195  varDecl(m);
196  Consume(tSemicolon);
197  }
198  else break;
199  }
200  }

204  subroutineDecl(m);

210  statseq = statSequence(m);

612 const CType* CParser::type(bool isParam, bool isFunction) {
613  //
614  // type ::= basetype | type "[" [ number ] "]".
615  // basetype ::= "boolean" | "char" | "integer".
616  //
617 
618  CToken t;
619  CToken tk;
620  EToken tt = _scanner->Peek().GetType();
621  CTypeManager *tm = CTypeManager::Get();
622  const CType *type;
623  vector<int> nelems;
624 
625  // basetype cases.
626  switch(tt) {
627  case tBoolean:
628  Consume(tBoolean, &t);
629  tk = t;
630  type = dynamic_cast<const CType *>(tm->GetBool());
631  break;
632  case tChar:
633  Consume(tChar, &t);
634  tk = t;
635  type = dynamic_cast<const CType *>(tm->GetChar());
636  break;
637  case tInteger:
638  Consume(tInteger, &t);
639  tk = t;
640  type = dynamic_cast<const CType *>(tm->GetInt());
641  break;
642 
643  default:
644  SetError(_scanner->Peek(), "basetype expected.");
645  break;
646  }
647 
648  assert(type != NULL);
649 
650  // array case.
651  while(_scanner->Peek().GetType() == tLSBrak) {
652  Consume(tLSBrak);
653  CToken nelem;
654 
655  if(_scanner->Peek().GetType() == tNumber) {
656  Consume(tNumber, &nelem);
657  nelems.push_back(atoi(nelem.GetValue().c_str()));
658  }
659  else {
660  // open arrays are allowed only in parameter.
661  if(isParam)
662  nelems.push_back(-1);

664  SetError(_scanner->Peek(), "expected 'tNumber', got '" + _scanner->Peek().GetName() + "'");
665  }
666  Consume(tRSBrak);
667  }
668

670  type = dynamic_cast<const CType *>(tm->GetArray(nelems[i], type));
671  assert(type != NULL);
672  nelems.pop_back();
673 
674  // Arrays are addressed in parameter.
675  if(i == 0 && isParam)
676  type = dynamic_cast<const CType *>(tm->GetPointer(type));
677  }
678 
679  if(isFunction && !type->IsScalar()) {

681  }

688  //
689  // statSequence ::= [ statement { ";" statement } ].
690  // FIRST(statement) = {tIF, tWhile, tReturn, tIdent}.
691  // FOLLOW(statSequence) = {tEnd, tElse}.
692  //
693 
694  CAstStatement *head = NULL;

696 
697  EToken tt = _scanner->Peek().GetType();

699  switch(tt) {

701  case tWhile:
702  case tReturn:
703  case tIdent:
704  head = tail = statement(s);
705  assert(head != NULL);
706 
707  // for multiple statements.
708  while(1) {
709  if(_scanner->Peek().GetType() == tEnd ||
710  _scanner->Peek().GetType() == tElse) {

712  }

714  Consume(tSemicolon);
715 
716  CAstStatement *st = NULL;
717  st = statement(s);
718 
719  assert(st != NULL);
720 
721  tail->SetNext(st);
722  tail = st;
723  }
724  }
725 
726  break;
727 
728  default:
729  break;
730  }
731 
732  return head;

737  // statement ::= assignment | subroutineCall | ifStatement | whileStatement | returnStatement.
738  // FIRST(statement) = {tIf, tWhile, tReturn, tIdent}.
